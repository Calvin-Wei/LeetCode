class Solution {
    public static final int RED = 1; // è¡¨ç¤ºçº¢è‰²
    public static final int GREEN = 2; // è¡¨ç¤ºç»¿è‰²
    public static final int UNCOLORED = 0; // è¡¨ç¤ºæœªæŸ“è‰²
    // æ–¹æ³•äºŒï¼š å¹¿åº¦ä¼˜å…ˆ
    public boolean possibleBipartition(int n, int[][] dislikes) {
        int color[] = new int[n];
        // å»ºå›¾
        List<List<Integer>> edge = new ArrayList();
        for(int i = 0; i < n; i++) {
            edge.add(new ArrayList());
        }
        for (int[] d : dislikes) {
            edge.get(d[0] - 1).add(d[1] - 1);
            edge.get(d[1] - 1).add(d[0] - 1);
        }
        for(int i = 0; i < n; i++) {
            if (color[i] == UNCOLORED) {
                Queue<Integer> queue = new LinkedList();
                queue.add(i);
                color[i] = RED;
                while (!queue.isEmpty()) {
                    Integer poll = queue.poll();
                    int reverse = color[poll] == RED ? GREEN : RED;
                    for (Integer u : edge.get(poll)) {
                        if (color[u] == UNCOLORED) {
                            color[u] = reverse;
                            queue.add(u);
                        }
                        if (color[u] != reverse) return false;
                    }
                }
            }
        }
        return true;
    }
}
