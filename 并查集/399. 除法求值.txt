class Solution {
    public double[] calcEquation(List<List<String>> equations, double[] values, List<List<String>> queries) {
        int equationSize = equations.size();
        UnionFind unionFind = new UnionFind(2 * equationSize);
        /**
         * ç¬¬ä¸€æ­¥ï¼šåšé¢„å¤„ç†
         * å°†å˜é‡çš„å€¼ ä¸ id è¿›è¡Œæ˜ å°„ï¼Œä½¿å¾—å¹¶æŸ¥é›†çš„åº•å±‚ä½¿ç”¨æ•°ç»„å®ç°ï¼Œæ–¹ä¾¿ç¼–ç 
         */
        Map<String, Integer> map = new HashMap<>(2 * equationSize);
        int id = 0;
        for(int i = 0;i < equationSize;i++){
            List<String> equation = equations.get(i);
            String var1 = equation.get(0);
            String var2 = equation.get(1);
            if(!map.containsKey(var1)){
                map.put(var1,id);
                id++;
            }
            if(!map.containsKey(var2)){
                map.put(var2,id);
                id++;
            }
            unionFind.union(map.get(var1), map.get(var2),values[i]);
        }
        /**
         * ç¬¬äºŒæ­¥ï¼šåšæŸ¥è¯¢
         */
        int queriesSize = queries.size();
        double [] res = new double[queriesSize];
        for(int i = 0;i < queriesSize;i++){
            String var1 = queries.get(i).get(0);
            String var2 = queries.get(i).get(1);

            Integer id1 = map.get(var1);
            Integer id2 = map.get(var2);

            if(id1 == null || id2 == null){
                res[i] = -1.0d;
            }else {
                res[i] = unionFind.isConnected(id1, id2);
            }
        }
        return res;
    }

    /**
     * å¹¶æŸ¥é›†
     */
    private class UnionFind{
        private int[] parent;

        /**
         * ç»“ç‚¹æŒ‡å‘çˆ¶èŠ‚ç‚¹çš„æƒå€¼
         */
        private double[] weight;

        public UnionFind(int n){
            this.parent = new int[n];
            this.weight = new double[n];
            for (int i = 0;i < n;i++){
                /**
                 * åˆå§‹åŒ–çš„æ—¶å€™
                 * é»˜è®¤çˆ¶äº²ç»“ç‚¹éƒ½æ˜¯å­é›† ä¸”æƒé‡ éƒ½ä¸º1ï¼
                 */
                parent[i] = i;
                weight[i] = 1.0d;
            }
        }

        public void union(int x,int y,double value){
            int rootX = find(x);
            int rootY = find(y);
            if(rootX == rootY){
                /**
                 * å¯¹åº”æ ¹èŠ‚ç‚¹ç›¸ç­‰
                 * åˆ™ä¸éœ€è¦ æ“ä½œæ ¹ç»“ç‚¹ï¼
                 */
                return;
            }
            /**
             * å¦åˆ™éœ€è¦å°†ä¸€ä¸ªç»“ç‚¹çš„æ ¹èŠ‚ç‚¹ æŒ‡å‘å¦ä¸€ä¸ªç»“ç‚¹çš„æ ¹èŠ‚ç‚¹ï¼
             */
            parent[rootX] = rootY;
            weight[rootX] = weight[y] * value / weight[x];
        }

        /**
         * è·¯å¾„å‹ç¼©
         */
        public int find(int x){
            if(x != parent[x]){
                int origin = parent[x];
                parent[x] = find(parent[x]);
                weight[x] *= weight[origin];
            }
            return parent[x];
        }

        public double isConnected(int x,int y){
            int rootX = find(x);
            int rootY = find(y);
            if(rootX == rootY){
                return weight[x] / weight[y];
            }else {
                return -1.0d;
            }
        }
    }
}
