class Edge {
    // åˆ†åˆ«è¡¨ç¤ºä¸€æ¡è¾¹ä¸¤ä¸ªç‚¹çš„ç´¢å¼•å’Œè¾¹é•¿
    int a, b, val;
    public Edge(int a, int b, int val) {
        this.a = a;
        this.b = b;
        this.val = val;
    }
}

class Solution {
    // å¹¶æŸ¥é›†é›†åˆ
    List<Integer> p;
    public int minCostConnectPoints(int[][] points) {
        p = new ArrayList<>();
        int n = points.length;
        // åˆå§‹åŒ–å¹¶æŸ¥é›†
        for(int i = 0; i < n; i++) {
            p.add(i);
        }
        // æ·»åŠ æ‰€æœ‰å¯èƒ½çš„è¾¹åˆ°åˆ—è¡¨ä¸­
        List<Edge> edges = new ArrayList<>();
        for(int i = 0; i < n - 1; i++) {
            for(int j = i + 1; j < n; j++) {
                // æ·»åŠ ç¬¬iä¸ªç‚¹å’Œç¬¬jä¸ªç‚¹ç»„æˆçš„è¾¹
                edges.add(new Edge(i, j, Math.abs(points[i][0] - points[j][0]) + Math.abs(points[i][1] - points[j][1])));
            }
        }
        // è¿›è¡Œä¸€ä¸ªå‡åºæ’åºï¼Œä¸ºäº†ä¼˜å…ˆé€‰å–çŸ­çš„è¾¹
        Collections.sort(edges, (a, b) -> a.val - b.val);

        // ä»ä½åˆ°é«˜éå†æ‰€æœ‰è¾¹
        int res = 0;
        for(Edge e : edges) {
            // æŸ¥æ‰¾è¿™æ¡è¾¹ä¸¤ä¸ªç«¯ç‚¹æ‰€å¤„çš„é›†åˆæ ¹å…ƒç´ 
            int a = find(e.a), b = find(e.b);
            // å¦‚æœè¿™ä¸¤æ¡è¾¹ä¸æ˜¯ä¸€ä¸ªé›†åˆçš„ï¼Œé‚£ä¹ˆå°±æ·»åŠ è¿™æ¡è¾¹
            if(a != b) {
                // å¹¶ä¸”å°†è¿™ä¸¤ä¸ªç‚¹æ‰€åœ¨çš„é›†åˆæ ¹å…ƒç´ åŠ å…¥ä¸€ä¸ªé›†åˆ
                p.set(a, b);
                // ç­”æ¡ˆä¸­æ·»åŠ è¿™æ¡è¾¹é•¿åº¦
                res += e.val;
            }
        }

        return res;
    }

    // å¹¶æŸ¥é›†æ¨¡æ¿æ–¹æ³•ï¼Œç”¨äºæŸ¥æ‰¾æ ¹å…ƒç´ 
    public int find(int x) {
        if(p.get(x) != x) {
            p.set(x, find(p.get(x)));
        }

        return p.get(x);
    }
}
