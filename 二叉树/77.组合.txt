class Solution {
    List<Integer> temp = new ArrayList<Integer>();
    List<List<Integer>> ans = new ArrayList<List<Integer>>();

    public List<List<Integer>> combine(int n, int k) {
        dfs(1, n, k);
        return ans;
    }

    public void dfs(int cur, int n, int k) {
        // å‰ªæï¼štemp é•¿åº¦åŠ ä¸ŠåŒºé—´ [cur, n] çš„é•¿åº¦å°äº kï¼Œä¸å¯èƒ½æ„é€ å‡ºé•¿åº¦ä¸º k çš„ temp
        if (temp.size() + (n - cur + 1) < k) {
            return;
        }
        // è®°å½•åˆæ³•çš„ç­”æ¡ˆ
        if (temp.size() == k) {
            ans.add(new ArrayList<Integer>(temp));
            return;
        }
        // è€ƒè™‘é€‰æ‹©å½“å‰ä½ç½®
        temp.add(cur);
        dfs(cur + 1, n, k);
        temp.remove(temp.size() - 1);
        // è€ƒè™‘ä¸é€‰æ‹©å½“å‰ä½ç½®
        dfs(cur + 1, n, k);
    }
}

class Solution {
    List<Integer> temp = new ArrayList<Integer>();
    List<List<Integer>> ans = new ArrayList<List<Integer>>();

    public List<List<Integer>> combine(int n, int k) {
        List<Integer> temp = new ArrayList<Integer>();
        List<List<Integer>> ans = new ArrayList<List<Integer>>();
        // åˆå§‹åŒ–
        // å°† temp ä¸­ [0, k - 1] æ¯ä¸ªä½ç½® i è®¾ç½®ä¸º i + 1ï¼Œå³ [0, k - 1] å­˜ [1, k]
        // æœ«å°¾åŠ ä¸€ä½ n + 1 ä½œä¸ºå“¨å…µ
        for (int i = 1; i <= k; ++i) {
            temp.add(i);
        }
        temp.add(n + 1);
        
        int j = 0;
        while (j < k) {
            ans.add(new ArrayList<Integer>(temp.subList(0, k)));
            j = 0;
            // å¯»æ‰¾ç¬¬ä¸€ä¸ª temp[j] + 1 != temp[j + 1] çš„ä½ç½® t
            // æˆ‘ä»¬éœ€è¦æŠŠ [0, t - 1] åŒºé—´å†…çš„æ¯ä¸ªä½ç½®é‡ç½®æˆ [1, t]
            while (j < k && temp.get(j) + 1 == temp.get(j + 1)) {
                temp.set(j, j + 1);
                ++j;
            }
            // j æ˜¯ç¬¬ä¸€ä¸ª temp[j] + 1 != temp[j + 1] çš„ä½ç½®
            temp.set(j, temp.get(j) + 1);
        }
        return ans;
    }
}
