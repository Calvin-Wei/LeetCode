class Solution {
    public int maxProduct(int[] nums) {
        if(nums==null || nums.length==0){
            return 0;
        }
        int max=nums[0];
        int min=nums[0];
        int maxEnd;
        int minEnd;
        int result=nums[0];
        for(int i=1;i<nums.length;i++){
            maxEnd=max*nums[i];
            minEnd=min*nums[i];
            max=Math.max(Math.max(maxEnd,minEnd),nums[i]);
            min=Math.min(Math.min(maxEnd,minEnd),nums[i]);
            result=Math.max(result,max);
        }
        return result;
    }
}
