class Solution {
    public int findTargetSumWays(int[] nums,int s){
        int sum=0;
        for(int item:nums){
            sum+=item;
        }
        if(Math.abs(s)>Math.abs(sum)){
            return 0;
        }
        int len=nums.length;
        int t=sum*2+1;
        int[][] dp=new int[len][t];
        if(nums[0]==0){
            dp[0][sum]=2;
        }else {
            dp[0][nums[0]+sum]=1;
            dp[0][sum-nums[0]]=1;
        }
        for(int i=1;i<len;i++){
            for(int j=0;j<t;j++){
                int l= Math.max((j - nums[i]), 0);
                int r= (j + nums[i]) < t ? j + nums[i] : 0;
                dp[i][j]=dp[i-1][l]+dp[i-1][r];
            }
        }
        return dp[len-1][sum+s];
    }
}
