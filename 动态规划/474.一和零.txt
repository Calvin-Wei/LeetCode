class Solution {
    public int findMaxForm(String[] strs, int m, int n) {
        //è¾“å…¥å­—ç¬¦ä¸²çš„å­åŒºé—´èƒ½å¤Ÿä½¿ç”¨jä¸ª0å’Œkä¸ª1çš„å­—ç¬¦ä¸²çš„æœ€å¤§æ•°é‡
        int[][] dp = new int[m + 1][n + 1];
        //è½¬ç§»æ–¹ç¨‹ï¼šä¸é€‰æ‹©å½“å‰ä¸ºdp[i][j]
        //é€‰æ‹©å½“å‰åˆ™ä¸ºdp[i-å½“å‰å­—ç¬¦ä¸²ä½¿ç”¨0çš„ä¸ªæ•°][j-å½“å‰å­—ç¬¦ä¸²ä½¿ç”¨1çš„ä¸ªæ•°]+1
        dp[0][0] = 0;
        for (String s : strs) {
            int[] zeroAndOne = calcZeroAndOne(s);
            int zeros = zeroAndOne[0];
            int ones = zeroAndOne[1];
            for (int i = m; i >= zeros; i--) {
                for (int j = n; j >= ones; j--) {
                    dp[i][j] = Math.max(dp[i][j], dp[i - zeros][j - ones] + 1);
                }
            }
        }
        return dp[m][n];
    }

    private int[] calcZeroAndOne(String str) {
        int[] res = new int[2];
        for (char c : str.toCharArray()) {
            res[c - '0']++;
        }
        return res;
    }
}
