class Solution {
    public int[] nextGreaterElements(int[] nums) {
        int[] array = new int[nums.length];
        Arrays.fill(array, -1);
        //å•è°ƒæ ˆä¸­å­˜æ”¾numsæ•°ç»„çš„ä¸‹æ ‡(å‰ä¸€ä¸ªå…ƒç´ çš„ä¸‹æ ‡)
        Stack<Integer> stack = new Stack<>();
        int len = nums.length;
        for (int i = 0; i < len * 2; i++) {
            //åˆ¤æ–­ä¸‹ä¸€ä¸ªå…ƒç´ æ˜¯å¦æ›´å¤§
            int temp = nums[i % len];
            //å¦‚æœå•è°ƒæ ˆä¸­çš„å…ƒç´ å°äºtemp,åˆ™å•è°ƒæ ˆå¼¹å‡ºå¹¶å†™å…¥ç»“æœä¸­
            while (!stack.isEmpty() && temp > nums[stack.peek()]) {
                array[stack.peek()] = temp;
                stack.pop();
            }
            stack.push(i % len);
        }
        return array;
    }
}
