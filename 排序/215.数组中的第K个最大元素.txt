class Solution {
    public int findKthLargest(int[] nums, int k) {
        int heapSize=nums.length;
        buildMaxHeap(nums,heapSize);
        for(int i=nums.length-1;i>=nums.length-k+1;i--){
            //å°†æœ€åä¸€ä¸ªå…ƒç´ æ”¾ç½®å †é¡¶,ç„¶åè¿›è¡Œæ’åº
            swap(nums,0,i);
            //heapSizeæœ€ç»ˆå˜ä¸ºlen-kä»£è¡¨ç¬¬kå¤§çš„æ•°æ”¾ç½®å †é¡¶åœ¨æ’åº
            --heapSize;
            //æ¯æ¬¡éƒ½é‡æ–°æ’åº
            maxHeapify(nums,0,heapSize);
        }
        return nums[0];
    }

    private void buildMaxHeap(int[] nums, int heapSize) {
        for (int i=heapSize/2;i>=0;i--){
            maxHeapify(nums,i,heapSize);
        }
    }

    private void maxHeapify(int[] nums, int index, int heapSize) {
        int left=index*2+1;
        while(left<heapSize){
            int largest=left+1<heapSize && nums[left+1] > nums[left] ? left+1:left;
            largest = nums[largest] < nums[index] ? index:largest;
            //æ²¡å˜åŒ–çš„æƒ…å†µ,è¡¨ç¤ºå †å·²å®Œæˆæ’åº,æå‰ç»ˆæ­¢
            if (largest==index){
                break;
            }
            swap(nums,index,largest);
            index=largest;
            left=index*2+1;
        }
    }

    private void swap(int[] nums, int i, int j) {
        int temp=nums[i];
        nums[i]=nums[j];
        nums[j]=temp;
    }
}
