class Solution {
    public List<List<Integer>> allPathsSourceTarget(int[][] graph) {
        List<List<Integer>> resultList = new ArrayList<>();
        dfs(graph, 0, graph.length, new LinkedList<>(), resultList);
        return resultList;
    }

    private void dfs(int[][] graph, int i, int length, LinkedList<Integer> list, List<List<Integer>> resultList) {
        if (i == length - 1) {
            list.add(i);
            resultList.add(new ArrayList<>(list));
            list.removeLast();
            return;
        }
        list.add(i);
        for (int node : graph[i]) {
            dfs(graph, node, length, list, resultList);
        }
        list.removeLast();
    }
}
