class Solution {
    int row, column;
    public void solve(char[][] board) {
        row = board.length;
        if (row == 0) {
            return;
        }
        column = board[0].length;
        //å…ˆéå†å››ä¸ªè¾¹ç•Œçš„å…ƒç´ ï¼Œå°†è¾¹ç•Œä¸ŠOçš„å…ƒç´ æ”¹ä¸ºAï¼Œç¬¬ä¸€åˆ—å’Œæœ€åä¸€åˆ—
        for (int i = 0; i < row; i++) {
            dfs(board,i,0);
            dfs(board,i,column-1);
        }
        //å…ˆéå†å››ä¸ªè¾¹ç•Œçš„å…ƒç´ ï¼Œå°†è¾¹ç•Œä¸ŠOçš„å…ƒç´ æ”¹ä¸ºAï¼Œç¬¬ä¸€åˆ—å’Œæœ€åä¸€åˆ—
        for(int i=1;i<column;i++){
            dfs(board,0,i);
            dfs(board,row-1,i);
        }
        //éå†æ¯ä¸ªäºŒç»´æ•°ç»„çš„æ¯ä¸€ä¸ªå…ƒç´ ,éå†æ¯ä¸€ä¸ªå…ƒç´ å°†Aæ”¹ä¸ºO,ç„¶åä¸­é—´æ²¡æœ‰è¢«å›´ç»•çš„Oæ”¹ä¸ºX
        for(int i=0;i<row;i++){
            for(int j=0;j<column;j++){
                if(board[i][j]=='A'){
                    board[i][j]='O';
                }else if(board[i][j]=='O'){
                    board[i][j]='X';
                }

            }
        }
    }

    private void dfs(char[][] board, int x, int y) {
        if (x < 0 || y < 0 || x >= row || y >= column || board[x][y] != 'O') {
            return;
        }
        //æ‰€æœ‰çš„Aéœ€è¦è¢«æ›¿æ¢ä¸ºX
        board[x][y] = 'A';
        dfs(board, x + 1, y);
        dfs(board, x - 1, y);
        dfs(board, x, y + 1);
        dfs(board, x, y - 1);
    }
}
