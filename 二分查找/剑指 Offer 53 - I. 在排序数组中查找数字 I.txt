class Solution {
    public int search(int[] nums, int target) {
        int left = 0, right = nums.length - 1;
        int count = 0;
        int mid = (right - left) / 2 + left;
        while (left <= right && mid <= right) {
            if (nums[mid] == target) {
                //æ‰¾åˆ°ç›®æ ‡å…ƒç´ ,å¾€å‰å’Œå¾€åéå†æŸ¥æ‰¾ç›®æ ‡å€¼è®¡ç®—æ•°é‡
                for (int i = mid; i >= 0; i--) {
                    if (nums[i] == target) {
                        count++;
                    }
                }
                for (int i = mid + 1; i <= right; i++) {
                    if (nums[i] == target) {
                        count++;
                    }
                }
                return count;
            } else if (nums[mid] > target) {
                right = mid - 1;
                mid = (right - left) / 2 + left;
            } else {
                left = mid + 1;
                mid = (right - left) / 2 + left;
            }
        }
        return count;
    }
}

class Solution {
    public int search(int[] nums, int target) {
        //æŸ¥æ‰¾ç›®æ ‡å…ƒç´ çš„ä½ç½®,æŸ¥æ‰¾æ¯”ç›®æ ‡å…ƒç´ -1çš„ä½ç½®,ä¸¤ä¸ªä½ç½®ç›¸å‡å°±æ˜¯é‡å¤å‡ºç°çš„æ¬¡æ•°
        return find(nums, target) - find(nums, target - 1);
    }

    private int find(int[] nums, int target) {
        int left = 0, right = nums.length - 1;
        while (left <= right) {
            int mid = (right - left) / 2 + left;
            if (nums[mid] <= target) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return left;
    }
}
