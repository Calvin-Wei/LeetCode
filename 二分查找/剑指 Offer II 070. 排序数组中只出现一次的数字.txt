class Solution {
    //äºŒåˆ†è§£æ³•
    public int singleNonDuplicate(int[] nums) {
        int index = -1, l = 0, r = nums.length - 1;
        while (l <= r) {
            int mid = (r - l) / 2 + l;
            //æ‰¾åˆ°midæ‰€åœ¨æ˜¯å³ä¾§æœ‰ä¸€ä¸ªç›¸åŒçš„æ•°è¿˜æ˜¯å·¦ä¾§æœ‰ä¸€ä¸ªç›¸åŒçš„æ•°
            if (mid + 1 < nums.length && nums[mid] == nums[mid + 1]) {
                //ä¸‹æ ‡ä¸ºå¶æ•°,å¥‡æ•°
                //å› ä¸ºé¢˜ç›®å›ºå®šæ¯ä¸ªæ•°æœ€å¤šé‡å¤ä¸¤æ¬¡,è®¡ç®—ä½ç½®èƒ½å¦è¢«2æ•´é™¤ï¼Œèƒ½è¢«2æ•´é™¤åˆ™è¯´æ˜å·¦æŒ‡é’ˆéœ€è¦+2ç§»åŠ¨åˆ°ä¸‹ä¸€ä¸ªæ•°
                if (mid % 2 == 0) {
                    l = mid + 2;
                } else {
                    r = mid - 1;
                }
            } else if (mid > 0 && nums[mid - 1] == nums[mid]) {
                //ä¸‹æ ‡ä¸ºå¥‡æ•°å¶æ•°
                if (mid % 2 == 0) {
                    r = mid - 2;
                } else {
                    l = mid + 1;
                }
            } else {
                index = nums[mid];
                break;
            }
        }
        return index;
    }
}
