import leetcode.BinarySearch;

/**
 * // This is MountainArray's API interface.
 * // You should not implement it, or speculate about its implementation
 * interface MountainArray {
 *     public int get(int index) {}
 *     public int length() {}
 * }
 */
 
class Solution {
    static class MountainArray {
        private int[] arr;
        private int length;

        public MountainArray(int[] arr, int length) {
            this.arr = arr;
            this.length = length;
        }

        public int get(int index) {
            return arr[index];
        }

        public int length() {
            return this.length;
        }
    }

    public int findInMountainArray(int target, BinarySearch.MountainArray mountainArr) {
        int n = mountainArr.length()-1;
        int top = binarySearchTop(mountainArr, 0, n);
        int left= binarySearchLeft(mountainArr,0,top,target);
        int right= binarySearchRight(mountainArr,top+1,n,target);
        return left==-1?right:left;
    }

    /**
     * å•è°ƒé€’å‡åŒºé—´
     */
    private int binarySearchRight(BinarySearch.MountainArray mountainArr, int left, int right, int target) {
        while(left<=right){
            int mid=(right-left)/2+left;
            int item=mountainArr.get(mid);
            if(target==item){
                return mid;
            }else if(item>target){
                left=mid+1;
            }else {
                right=mid-1;
            }
        }
        return -1;
    }

    /**
     * å•è°ƒé€’å¢åŒºé—´
     */
    private int binarySearchLeft(BinarySearch.MountainArray mountainArr, int left, int right, int target) {
        while(left<=right){
            int mid=left+(right-left)/2;
            int item=mountainArr.get(mid);
            if(item==target){
                return mid;
            }else if(item>target){
                right=mid-1;
            }else {
                left=mid+1;
            }
        }
        return -1;
    }

    /**
     * æŸ¥æ‰¾å³°é¡¶æ‰€åœ¨ä½ç½®
     */
    private int binarySearchTop(BinarySearch.MountainArray mountainArr, int left, int right) {
        while (left <= right) {
            int mid = left + (right - left) / 2;
            if (mountainArr.get(mid) < mountainArr.get(mid + 1)) {
                left = mid + 1;
            } else {
                right = mid - 1;
            }
        }
        return left;
    }
}
