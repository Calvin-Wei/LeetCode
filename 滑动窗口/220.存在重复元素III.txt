class Solution {
    public boolean containsNearbyAlmostDuplicate(int[] nums, int k, int t) {
        //ç»´æŠ¤ké•¿åº¦çš„æ»‘åŠ¨çª—å£,åˆ¤æ–­åŒºé—´ä¸ºnums[i]+t,nums[i]-t
        //å®ç°æ–¹é¢ï¼Œæˆ‘ä»¬åœ¨æ»‘åŠ¨çª—å£ä¸­æŸ¥æ‰¾å¤§äºç­‰äº xâˆ’t çš„æœ€å°çš„å…ƒç´  yï¼Œå¦‚æœ y å­˜åœ¨ï¼Œä¸” yâ‰¤x+tï¼Œæˆ‘ä»¬å°±æ‰¾åˆ°äº†ä¸€å¯¹ç¬¦åˆæ¡ä»¶çš„å…ƒç´ ã€‚
        // å®Œæˆæ£€æŸ¥åï¼Œæˆ‘ä»¬å°† x æ’å…¥åˆ°æ»‘åŠ¨çª—å£ä¸­ï¼Œå¦‚æœæœ‰åºé›†åˆä¸­å…ƒç´ æ•°é‡è¶…è¿‡äº† kï¼Œæˆ‘ä»¬å°†æ»‘åŠ¨çª—å£ä¸­æœ€æ—©è¢«æ’å…¥çš„å…ƒç´ åˆ é™¤å³å¯ã€‚
        TreeSet<Long> set = new TreeSet<>();
        for (int i = 0; i < nums.length; i++) {
            //ç›¸å½“äºæŸ¥æ‰¾nums[j]>=abs(nums[i]-t),å¦‚æœæ‰¾åˆ°åˆ™åˆ¤æ–­æ˜¯å¦æ¯”num[i]+tå°
            Long ceiling = set.ceiling((long) nums[i] - (long) t);
            if (ceiling != null && ceiling <= (long) nums[i] + (long) t) {
                return true;
            }
            set.add((long) nums[i]);
            if (i >= k) {
                set.remove((long) nums[i - k]);
            }
        }
        return false;
    }
}
