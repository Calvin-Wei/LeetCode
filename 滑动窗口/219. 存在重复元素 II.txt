class Solution {
    public boolean containsNearbyDuplicate(int[] nums, int k) {
        HashSet<Integer> set=new HashSet<>();
        //ç»´æŠ¤ä¸€ä¸ªkå¤§å°çš„æ»‘åŠ¨çª—å£,ç„¶åä¸€ç›´ç§»åŠ¨
        for(int i=0;i<nums.length;i++){
            if(set.contains(nums[i]))
                return true;
            set.add(nums[i]);
            //è¶…è¿‡k,åˆ™æŠŠæ»‘åŠ¨çª—å£çš„ç¬¬ä¸€ä¸ªå…ƒç´ ç§»é™¤
            if(set.size()>k){
                set.remove(nums[i-k]);
            }
        }
        return false;
    }
}
