class Solution {
    public int minSubArray(int target, int[] nums) {
        int min = nums.length;
        //çª—å£çš„å·¦å³è¾¹ç•Œ
        int left = 0;
        //å½“å‰å…ƒç´ çš„å’Œ
        int sum = 0;
        boolean flag = false;
        for (int right = 0; right < nums.length; right++) {
            sum += nums[right];
            //å¦‚æœå½“å‰å…ƒç´ å’Œ>=target
            while (sum >= target) {
                //å–ä¹‹å‰çª—å£é•¿åº¦ä¸å½“å‰çª—å£é•¿åº¦æœ€çŸ­çš„
                min = Math.min(min, right - left + 1);
                //å»æ‰å·¦ä¾§æ•°æ®
                sum -= nums[left];
                //ç§»åŠ¨çª—å£çš„å·¦æŒ‡é’ˆ
                left++;
                flag = true;
            }
        }
        return flag ? min : 0;
    }
}
