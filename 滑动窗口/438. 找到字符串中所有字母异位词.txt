class Solution {
    public List<Integer> findAnagrams(String s, String p) {
        List<Integer> ans=new ArrayList<>();
        if(s.length()<p.length()){
            return ans;
        }
        int[] sCnt =new int[26];
        int[] pCnt =new int[26];
        for(int i=0;i<p.length();i++){
            pCnt[p.charAt(i)-'a']++;
        }
        //indexæŒ‡é’ˆæ‰«æçš„æ˜¯så­—ç¬¦ä¸²å¼€å§‹ç‚¹ä½ç½®
        int index=0;
        //iæ˜¯ä¸€ä¸ªä¸ªæ‰«æså­—ç¬¦ä¸²
        for(int i=0;i<s.length();i++){
            int cur=s.charAt(i)-'a';
            //å­—ç¬¦çš„å‡ºç°è®¡æ•°
            sCnt[cur]++;
            //å¦‚æœå½“å‰så­—ç¬¦ä¸²çš„å­—ç¬¦æ²¡æœ‰åœ¨pä¸­å‡ºç°,æˆ–è€…å‡ºç°æ¬¡æ•°å¤§äºpä¸­çš„æ•°é‡
            //åˆ™å°†så­—ç¬¦ä¸²å’Œpå­—ç¬¦ä¸²ä¸å¯¹åº”åŒºé—´è¿›è¡Œindexç§»åŠ¨ï¼Œå¹¶å°†è®¡æ•°æ¸…é™¤
            while(sCnt[cur]>pCnt[cur]){
                sCnt[s.charAt(index)-'a']--;
                index++;
            }
            //é•¿åº¦åŒ¹é…på­—ç¬¦ä¸²åˆ™ä¸ºç»“æœ
            if(i-index+1==p.length()){
                ans.add(index);
            }
        }
        return ans;
    }
}
