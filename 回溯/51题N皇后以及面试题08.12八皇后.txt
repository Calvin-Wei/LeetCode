class Solution{
    List<List<String>> result=new ArrayList<>();

    public List<List<String>> solveNQueens(int n){
        char[][] chessboard=new char[n][n];
        for(char[] array:chessboard){
            Arrays.fill(array,'.');
        }
        backTrack(n,0,chessboard);
        return result;
    }

    private void backTrack(int n, int row, char[][] chessboard) {
        if(row==n){
            result.add(ArrayToList(chessboard));
            return;
        }
        for(int col=0;col<n;col++){
            //éªŒè¯åˆæ³•
            if(isValid(row,col,n,chessboard)){
                //æ”¾ç½®çš‡å
                chessboard[row][col]='Q';
                //å› ä¸ºæ¯ä¸€è¡Œåªä¼šæ”¾ç½®ä¸€ä¸ªå…ƒç´ ,ä¸ç”¨æ ¡éªŒè¡Œ
                backTrack(n,row+1,chessboard);
                //å›æº¯ï¼Œæ’¤é”€çš‡å
                chessboard[row][col]='.';
            }
        }
    }

    private boolean isValid(int row, int col, int n, char[][] chessboard) {
        //åˆ¤æ–­åˆ—,ç›¸å½“äºå‰ªæ
        for(int i=0;i<row;i++){
            if(chessboard[i][col]=='Q'){
                return false;
            }
        }
        //å·¦ä¾§å¯¹è§’çº¿ï¼Œæ ¡éªŒå…ƒç´ çš„å·¦ä¸Šä¾§æœ‰æ— å…ƒç´ 
        for(int i=row-1,j=col-1;i>=0 && j>=0;i--,j--){
            if(chessboard[i][j]=='Q'){
                return false;
            }
        }
        //å³ä¾§å¯¹è§’çº¿ï¼Œæ ¡éªŒå…ƒç´ çš„å³ä¸Šä¾§æœ‰æ— å…ƒç´ 
        for(int i=row-1,j=col+1;i>=0 && j<=n-1;i--,j++){
            if(chessboard[i][j]=='Q'){
                return false;
            }
        }
        return true;
    }

    private List<String> ArrayToList(char[][] chessboard) {
        List<String> list=new ArrayList<>();
        for(char[] c:chessboard){
            list.add(String.copyValueOf(c));
        }
        return list;
    }
}
