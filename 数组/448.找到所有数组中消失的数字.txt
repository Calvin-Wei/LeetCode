class Solution {

    public List<Integer> findDisappearedNumbers(int[] nums) {
        List<Integer> resultList = new ArrayList<>();
        int i=0;
        while(i < nums.length) {
            //ä¸€ä¸ªèåœä¸€ä¸ªå‘,ä¾‹å¦‚ï¼šæ•°å­—1åœ¨0
            if (i + 1 == nums[i]) {
                i++;
                continue;
            }
            //å½“å‰å…ƒç´ åº”è¯¥å‡ºç°åœ¨å“ªä¸ªä½ç½®,ä¾‹å¦‚8åº”è¯¥å‡ºç°åœ¨7ä½ç½®
            int index = nums[i] - 1;
            //å½“å‰å…ƒç´ =ç†æƒ³ä½ç½®ä¸Šçš„ç°æœ‰å…ƒç´ ,è¯´æ˜æ•°ç»„ä¸­æœ‰é‡å¤æ•°å­—
            if (nums[i] == nums[index]) {
                i++;
                continue;
            }
            //ä¸é‡å¤åˆ™äº¤æ¢.è¿™é‡Œä¸è¿›è¡Œi++æ˜¯å› ä¸ºäº¤æ¢è¿‡æ¥çš„å…ƒç´ ä¹Ÿéœ€è¦åˆ¤æ–­æ˜¯ä¸æ˜¯åœ¨ç†æƒ³ä½ç½®ä¸Š
            swap(nums, i, index);
        }
        for (int j = 0; j < nums.length; j++) {
            if (nums[j] != j + 1) {
                resultList.add(j + 1);
            }
        }
        return resultList;
    }
    private void swap(int[] nums, int i, int j) {
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
}
